\section*{Podział prac według punktów kontrolnych}

\subsection*{Wojciech Jasiński (koordynator)}
\begin{itemize}\setlength\itemsep{0.1em}
  \item \textbf{PK1:} koordynacja zespołu, ustalenie zakresu klasy $\mathbb{DS4}$ i~struktury dokumentu teorii wstępnej.
  \item \textbf{PK2:} złożenie pełnej teorii w~\LaTeX{}, spójność notacji, przygotowanie wydruku.
  \item \textbf{PK3:} szkielet rozwiązania (projekty \texttt{Ds4.Core}/\texttt{Cli}/\texttt{Gui}), integracja modułów, zestaw przykładów demonstracyjnych.
  \item \textbf{PK4:} reorganizacja teorii wg~uwag, dokument techniczny, kompilacja i~wydanie aplikacji, scalenie gałęzi.
\end{itemize}

\subsection*{Krzysztof Gryboś}
\begin{itemize}\setlength\itemsep{0.1em}
  \item \textbf{PK1:} wstępny opis języka kwerend i~rodzajów pytań (\textit{executable}, cel).
  \item \textbf{PK2:} formalna semantyka \textit{necessary}/\textit{possibly} dla wielu stanów początkowych, proces i~ślad wykonania.
  \item \textbf{PK3:} silnik procesów i~kwerend (drzewo wykonań, ewaluacja czterech wariantów pytań).
  \item \textbf{PK4:} korekta semantyki \textit{possibly} (kwantyfikacja uniwersalna po~$\Sigma_0$), testy procesów i~kwerend.
\end{itemize}
